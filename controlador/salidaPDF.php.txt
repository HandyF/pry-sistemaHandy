<?php
require_once 'controlador/conecta.php';
$producto = 'SELECT
r.ID_RECEPCION as Identificador, e.NOMBRES_EMPLEADO + e.APELLIDOS_EMPLEADO as EmpleadoNombresyApellidos, td.DESCRIPCION, r.FECHA_HORA, r.NUMERO_TIPO_DOCUMENTO

 FROM  trecepcion r inner join templeado e on r.ID_EMPLEADO= e.ID_EMPLEADO
       join ttipo_documento td on r.ID_TIPO_DOCUMENTO= td.ID_TIPO_DOCUMENTO';
$productos = $mysqli->query($producto);

if (isset($_POST['create_pdf'])) {
    require_once 'tcpdf/tcpdf.php';

    $pdf = new TCPDF('P', 'mm', 'A4', true, 'UTF-8', false);

    $pdf->SetCreator(PDF_CREATOR);
    $pdf->SetAuthor('Handy Andres Fierro Ortiz');
    $pdf->SetTitle($_POST['reporte_name']);

    $pdf->setPrintHeader(false);
    $pdf->setPrintFooter(false);
    $pdf->SetMargins(20, 20, 20, false);
    $pdf->SetAutoPageBreak(true, 20);
    $pdf->SetFont('Helvetica', '', 10);
    $pdf->addPage();

    $content = '';

    $content .= '
        <div class="row">
            <div class="col-sm-8">
                <h1 style="text-align:center;">' . $_POST['reporte_name'] . '</h1>

      <table border="1" cellpadding="5">

          <tr>
            <th>Nombres y Apellidos del Empleado</th>
            <th>Tipo Documento</th>
            <th>Fecha recepcion</th>
            <th>Nº del Tipo Documento</th>
          </tr>

    ';

    while ($produc = $productos->fetch_assoc()) {
        $content .= '
        <tr>
            <td>' . $produc['EmpleadoNombresyApellidos'] . '</td>
            <td>' . $produc['DESCRIPCION'] . '</td>
            <td>' . $produc['FECHA_HORA'] . '</td>
            <td>' . $produc['NUMERO_TIPO_DOCUMENTO'] . '</td>
        </tr>
    ';
    }

    $content .= '</table>';

    $content .= '
        <div class="row padding">
            <div class="col-sm-6" style="text-align:center;">
                <span>Pdf Creator </span><a href="http://www.redecodifica.com">By Handy Andres Fierro Ortiz</a>
            </div>
        </div>

    ';

    $pdf->writeHTML($content, true, 0, true, 0);

    $pdf->lastPage();
    $pdf->output('Reporte.pdf', 'I');
}
